71 void StatsdClient::send (const std::string& key, const int value, const std::string& type, const float frequency) const
72  noexcept
73 {
74  const auto isFrequencyOne = [](const float freq) noexcept
75  {
76  constexpr float epsilon{0.0001f};
77  return std::fabs (freq - 1.0f) < epsilon;
78  };
79 
80  // Test if one should send or not, according to the frequency rate
81  if (!isFrequencyOne (frequency))
82  {
83  if (frequency < static_cast<float> (std::rand ()) / static_cast<float> (RAND_MAX))
84  {
85  return;
86  }
87  }
88 
89  // Prepare the buffer, with a sampling rate if specified different from 1.0f
90  char buffer[256];
91  if (isFrequencyOne (frequency))
92  {
93  // Sampling rate is 1.0f, no need to specify it
94  std::snprintf (buffer, sizeof (buffer), "%s%s:%d|%s", m_prefix.c_str (), key.c_str (), value, type.c_str ());
95  }
96  else
97  {
98  // Sampling rate is different from 1.0f, hence specify it
99  std::snprintf (buffer,
100  sizeof (buffer),
101  "%s%s:%d|%s|@%.2f",
102  m_prefix.c_str (),
103  key.c_str (),
104  value,
105  type.c_str (),
106  static_cast<double> (frequency));
107  }
108 
109  // Send the message via the UDP sender
110  m_sender.send (buffer);
111 }
