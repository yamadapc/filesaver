89 void InMemoryFileEntryStore::updatePendingAndFinishedState (boost::filesystem::path filepath)
90 {
91  const auto& filepathStr = filepath.string ();
92  auto record = m_records[filepathStr];
93  auto entry = record.fileEntry;
94  entry->isFinished = true;
95 
96  if (m_delegate.has_value ())

101  filepath.remove_filename ();
